// JavaScript Program for ADD, SUB, MULTI using switch case let num1 = parseFloat(prompt("Enter first number:")); let num2 = parseFloat(prompt("Enter second number:")); let operation = prompt("Enter operation (ADD, SUB, MULTI):").toUpperCase(); let result; switch(operation) { case "ADD": result = num1 + num2; console.log("Addition: " + result); break; case "SUB": result = num1 - num2; console.log("Subtraction: " + result); break; case "MULTI": result = num1 * num2; console.log("Multiplication: " + result); break; default: console.log("Invalid operation selected!"); } alert("Result: " + result);
